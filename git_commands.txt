git clone ссылка_на_ваш_репозиторий (клонирует репозиторий по ссылке)

git add . (добавляет все измененные файлы в коммит)
git commit -m "название_коммита" (настройка коммита)
git push origin "название_бранча" (запушить коммит в бранч)

git config --global user.name "ваше_имя" (указать ваш юзернейм от гитхаба)
git config --global user.email ваша_почта (указать вашу почту от гитхаба)
